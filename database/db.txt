@startuml

entity Users {
    + id : integer [PK]
    --
    name : varchar
    email : varchar [unique]
    email_verified_at : timestamp
    password : varchar
    remember_token : varchar
    created_at : timestamp
    updated_at : timestamp
}

entity Banks {
    + id : integer [PK]
    --
    name : varchar
    bank : varchar
    norek : varchar
    image : varchar
    user_id : integer [FK]
    created_at : timestamp
    updated_at : timestamp
}

entity Categories {
    + id : integer [PK]
    --
    name : varchar
    user_id : integer [FK]
    pengeluaran : boolean
    image : varchar
    created_at : timestamp
    updated_at : timestamp
}

entity Transactions {
    + id : integer [PK]
    --
    category_id : integer [FK]
    user_id : integer [FK]
    bank_id : integer [FK]
    date : date
    amount : integer
    description : varchar
    image : varchar
    created_at : timestamp
    updated_at : timestamp
}

entity Mybios {
    + id : integer [PK]
    --
    user_id : integer [FK, unique]
    birth : date
    gender : varchar
    address : varchar
    image : varchar
    created_at : timestamp
    updated_at : timestamp
}

// Relationships
Users ||--o{ Banks : "has"
Users ||--o{ Categories : "has"
Users ||--o{ Transactions : "has"
Categories ||--o{ Transactions : "contains"
Banks ||--o{ Transactions : "used in"
Users ||--|| Mybios : "has (One-to-One)"

@enduml
